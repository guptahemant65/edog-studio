```javascript
  /**
   * FLT fetch with structured errors (throws on any error, never returns null).
   * @param {string} path
   * @param {object} [options]
   * @returns {Promise<object>}
   * @throws {Error} With .status, .body, .path
   */
  async _fltFetchStrict(path, options = {}) {
    if (!this._fabricBaseUrl || !this._mwcToken) {
      const err = new Error('FLT service not connected');
      err.status = 0; err.path = path; throw err;
    }
    const resp = await fetch(this._fabricBaseUrl + path, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `MwcToken ${this._mwcToken}`,
        ...options.headers
      },
    });
    if (!resp.ok) {
      const err = new Error(`FLT API error: ${resp.status} ${path}`);
      err.status = resp.status;
      err.body = await resp.text().catch(() => '');
      err.path = path;
      throw err;
    }
    const text = await resp.text();
    return text ? JSON.parse(text) : {};
  }

  /**
   * FLT fetch returning raw Response (for reading headers like continuation tokens).
   */
  async _fltFetchRaw(path, options = {}) {
    if (!this._fabricBaseUrl || !this._mwcToken) throw new Error('FLT not connected');
    const resp = await fetch(this._fabricBaseUrl + path, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `MwcToken ${this._mwcToken}`,
      },
    });
    if (!resp.ok) {
      const err = new Error(`FLT API error: ${resp.status} ${path}`);
      err.status = resp.status; err.path = path; throw err;
    }
    return resp;
  }
```

- [ ] **Step 1: Create `dag-layout.js` with DagLayout class**

```javascript
/**
 * DAG Layout Engine — Sugiyama layered algorithm.
 *
 * Input:  nodes[] (id, name, kind) + edges[] (from, to)
 * Output: positioned nodes[] (+ x, y, w, h) + routed edges[]
 *
 * Five steps: layer assignment → dummy insertion → crossing minimization →
 * coordinate assignment → edge routing.
 *
 * Complexity: O(V² + E) — < 50ms for 50 nodes, < 500ms for 300 nodes.
 */
class DagLayout {
  constructor(options = {}) {
    this.layerSpacing = options.layerSpacing || 200;
    this.nodeSpacing = options.nodeSpacing || 80;
    this.nodeWidth = options.nodeWidth || 160;
    this.nodeHeight = options.nodeHeight || 56;
    this.edgeControlOffset = options.edgeControlOffset || 40;
  }

  /**
   * Run full layout pipeline.
   * @param {Array<{id:string, name:string, kind:string}>} nodes
   * @param {Array<{from:string, to:string}>} edges
   * @returns {{nodes: Array, edges: Array, layers: number, bounds: {x,y,w,h}}}
   */
  layout(nodes, edges) {
    if (!nodes.length) return { nodes: [], edges: [], layers: 0, bounds: { x: 0, y: 0, w: 0, h: 0 } };

    const adj = this._buildAdjacency(nodes, edges);
    const layers = this._assignLayers(nodes, adj);
    const layerMap = this._groupByLayer(nodes, layers);
    const maxLayer = Math.max(...Object.values(layers));

    // Insert dummy nodes for edges spanning multiple layers
    const { allNodes, allEdges } = this._insertDummies(nodes, edges, layers);
    const dummyLayers = this._assignLayers(allNodes, this._buildAdjacency(allNodes, allEdges));
    const dummyLayerMap = this._groupByLayer(allNodes, dummyLayers);

    // Crossing minimization (barycenter heuristic, 2-pass)
    this._minimizeCrossings(dummyLayerMap, allEdges, dummyLayers, maxLayer);

    // Coordinate assignment
    this._assignCoordinates(allNodes, dummyLayers, dummyLayerMap);

    // Filter out dummy nodes, keep real nodes with positions
    const positioned = allNodes.filter(n => !n._dummy).map(n => ({
      ...n, w: this.nodeWidth, h: this.nodeHeight
    }));

    // Route edges
    const routed = this._routeEdges(edges, allNodes, dummyLayers);

    // Calculate bounds
    const bounds = this._calculateBounds(positioned);

    return {
      nodes: positioned,
      edges: routed,
      layers: maxLayer + 1,
      bounds,
    };
  }

  /** Step 1: Kahn's topological sort for layer assignment. O(V + E). */
  _assignLayers(nodes, adj) {
    const inDegree = new Map();
    const layers = {};
    nodes.forEach(n => { inDegree.set(n.id, 0); layers[n.id] = 0; });

    for (const [, targets] of adj.forward) {
      for (const t of targets) {
        inDegree.set(t, (inDegree.get(t) || 0) + 1);
      }
    }

    const queue = [];
    for (const [id, deg] of inDegree) {
      if (deg === 0) queue.push(id);
    }

    while (queue.length) {
      const nodeId = queue.shift();
      const targets = adj.forward.get(nodeId) || [];
      for (const t of targets) {
        layers[t] = Math.max(layers[t], layers[nodeId] + 1);
        inDegree.set(t, inDegree.get(t) - 1);
        if (inDegree.get(t) === 0) queue.push(t);
      }
    }

    return layers;
  }

  /** Build forward + reverse adjacency maps. */
  _buildAdjacency(nodes, edges) {
    const forward = new Map();
    const reverse = new Map();
    nodes.forEach(n => { forward.set(n.id, []); reverse.set(n.id, []); });
    edges.forEach(e => {
      if (forward.has(e.from)) forward.get(e.from).push(e.to);
      if (reverse.has(e.to)) reverse.get(e.to).push(e.from);
    });
    return { forward, reverse };
  }

  /** Group node IDs by layer. */
  _groupByLayer(nodes, layers) {
    const map = {};
    nodes.forEach(n => {
      const l = layers[n.id];
      if (!map[l]) map[l] = [];
      map[l].push(n.id);
    });
    return map;
  }

  /** Step 2: Insert dummy nodes for multi-layer edges. O(E). */
  _insertDummies(nodes, edges, layers) {
    const allNodes = nodes.map(n => ({ ...n }));
    const allEdges = [];
    let dummyCount = 0;

    for (const e of edges) {
      const fromLayer = layers[e.from];
      const toLayer = layers[e.to];
      const span = toLayer - fromLayer;

      if (span <= 1) {
        allEdges.push({ ...e });
      } else {
        let prev = e.from;
        for (let i = 1; i < span; i++) {
          const dummyId = `__dummy_${dummyCount++}`;
          allNodes.push({ id: dummyId, name: '', _dummy: true, _edgeFrom: e.from, _edgeTo: e.to });
          allEdges.push({ from: prev, to: dummyId });
          prev = dummyId;
        }
        allEdges.push({ from: prev, to: e.to });
      }
    }

    return { allNodes, allEdges };
  }

  /** Step 3: Barycenter crossing minimization. 2-pass. O(V² per pass). */
  _minimizeCrossings(layerMap, edges, layers, maxLayer) {
    const adj = this._buildAdjacency(
      Object.values(layerMap).flat().map(id => ({ id })),
      edges
    );

    // Top-down pass
    for (let l = 1; l <= maxLayer; l++) {
      if (!layerMap[l]) continue;
      const barycenters = {};
      for (const nodeId of layerMap[l]) {
        const parents = (adj.reverse.get(nodeId) || []).filter(p => layers[p] === l - 1);
        if (parents.length && layerMap[l - 1]) {
          const positions = parents.map(p => layerMap[l - 1].indexOf(p)).filter(i => i >= 0);
          barycenters[nodeId] = positions.length ? positions.reduce((a, b) => a + b, 0) / positions.length : 0;
        } else {
          barycenters[nodeId] = layerMap[l].indexOf(nodeId);
        }
      }
      layerMap[l].sort((a, b) => barycenters[a] - barycenters[b]);
    }

    // Bottom-up pass
    for (let l = maxLayer - 1; l >= 0; l--) {
      if (!layerMap[l]) continue;
      const barycenters = {};
      for (const nodeId of layerMap[l]) {
        const children = (adj.forward.get(nodeId) || []).filter(c => layers[c] === l + 1);
        if (children.length && layerMap[l + 1]) {
          const positions = children.map(c => layerMap[l + 1].indexOf(c)).filter(i => i >= 0);
          barycenters[nodeId] = positions.length ? positions.reduce((a, b) => a + b, 0) / positions.length : 0;
        } else {
          barycenters[nodeId] = layerMap[l].indexOf(nodeId);
        }
      }
      layerMap[l].sort((a, b) => barycenters[a] - barycenters[b]);
    }
  }

  /** Step 4: Assign x/y coordinates based on layer + position. O(V). */
  _assignCoordinates(allNodes, layers, layerMap) {
    const nodeMap = new Map(allNodes.map(n => [n.id, n]));

    for (const [layerStr, nodeIds] of Object.entries(layerMap)) {
      const layer = parseInt(layerStr);
      const count = nodeIds.length;
      nodeIds.forEach((id, pos) => {
        const node = nodeMap.get(id);
        if (!node) return;
        node.x = layer * this.layerSpacing;
        node.y = pos * this.nodeSpacing - (count * this.nodeSpacing / 2) + this.nodeSpacing / 2;
      });
    }
  }

  /** Step 5: Route edges as bezier curves. O(E). */
  _routeEdges(originalEdges, allNodes, layers) {
    const nodeMap = new Map(allNodes.map(n => [n.id, n]));

    return originalEdges.map(e => {
      const from = nodeMap.get(e.from);
      const to = nodeMap.get(e.to);
      if (!from || !to) return null;

      // Collect waypoints (dummy nodes along this edge)
      const waypoints = allNodes
        .filter(n => n._dummy && n._edgeFrom === e.from && n._edgeTo === e.to)
        .sort((a, b) => (layers[a.id] || 0) - (layers[b.id] || 0));

      const points = [
        { x: from.x + this.nodeWidth, y: from.y + this.nodeHeight / 2 },
        ...waypoints.map(w => ({ x: w.x + this.nodeWidth / 2, y: w.y + this.nodeHeight / 2 })),
        { x: to.x, y: to.y + this.nodeHeight / 2 },
      ];

      return { from: e.from, to: e.to, points };
    }).filter(Boolean);
  }

  /** Calculate bounding box of all positioned nodes. */
  _calculateBounds(nodes) {
    if (!nodes.length) return { x: 0, y: 0, w: 0, h: 0 };
    let minX = Infinity, minY = Infinity, maxX = -Infinity, maxY = -Infinity;
    for (const n of nodes) {
      minX = Math.min(minX, n.x);
      minY = Math.min(minY, n.y);
      maxX = Math.max(maxX, n.x + n.w);
      maxY = Math.max(maxY, n.y + n.h);
    }
    return { x: minX, y: minY, w: maxX - minX, h: maxY - minY };
  }
}
```
